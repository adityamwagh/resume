% Bag of Visual Words for finding similar images
\resumeItem{Bag of Visual Words for finding similar images}{(opencv, scikit-learn)}
\resumeItemListStart
\resumeItem{Introduction}
{Developed a method to find similar query images from a database of 29,000 50x50 resolution images.}
\resumeItem{Feature Extraction}{Computed SIFT features for each image in database using OpenCV's built-in SIFT feature extractor.}
\resumeItem{Clustering}{Used the k-means clustering algorithm to compute 800 cluster centroids which serve as visual words.}
\resumeItem{Histograms}{Computed histograms of computed visual words for all the query images and images in the database.}
\resumeItem{Image Query}{Used the k-nearest neighbours algorithm to find the histogram that is most similar to query image.}
\resumeItemListEnd
